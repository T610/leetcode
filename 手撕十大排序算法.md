# 冒泡排序

```python
def swap(i,j):
            temp = num[i]
            num[i] = num[j]
            num[j] = temp
        for i in range(len(num)):      
            for j in range(len(num)-i-1):  
                if num[j] > num[j+1] :   
                    swap(j,j+1)          
```

## 快速排序

```python
import random
 
class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        def swap(i,j) :
            temp = nums[i]
            nums[i] = nums[j]
            nums[j] = temp

        def para(left,right):
            mid = random.randint(left,right)
            swap( left , mid )
            para = nums[left] 
            i = left + 1
            j = left + 1
            while i <= right  :
                if nums[i] <= para  :
                    swap(i,j)
                    j += 1
                i += 1
            swap(left,j-1)
            return j-1

        def curr(left,right) :  
            if right <= left :  
                return 
            mid = para(left,right)
            curr(left,mid-1) 
            curr(mid+1,right) 
        curr(0 , len(nums) - 1)
        return nums
```
